package com.example.Atividades;

import android.app.Application;
import android.content.Intent;
import android.os.Bundle;
import android.support.v7.app.AppCompatActivity;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;

import java.text.DateFormat;
import java.text.ParseException;
import java.text.SimpleDateFormat;
import java.util.Date;

import io.realm.Realm;
import io.realm.RealmConfiguration;
import io.realm.RealmResults;
import io.realm.annotations.Required;

import com.example.tp_pdmm.model.Curso;
import com.example.tp_pdmm.model.*;
public class ActividadeCurso extends AppCompatActivity {
    Button add, view, update, delete;
    EditText nomedocurso, Descrição,iddocurso;
    TextView text;
    Realm realm;

    @Override
    public void onCreate(Bundle savedInstanceState){

        super.onCreate(savedInstanceState);
        setContentView(R.layout.curso_layout);



        add = (Button)findViewById(R.id.add);
        view = (Button)findViewById(R.id.view);
        update = (Button)findViewById(R.id.update);
        delete = (Button)findViewById(R.id.delete);
        text = (TextView)findViewById(R.id.text);
        Descrição = (EditText)findViewById(R.id.Descricao);
        nomedocurso = (EditText)findViewById(R.id.nomedocurso);
        iddocurso = (EditText)findViewById(R.id.idcurso);


        Realm.init(this);
        realm = Realm.getDefaultInstance();
    }
    public void clickAction(View view){
        switch (view.getId()){
            case R.id.add:  addRecord();
                break;
            case R.id.view: viewRecord();
                break;
            case R.id.update:   updateRecord();
                break;
            case R.id.delete:   deleteRecord();
                break;

        }
    }
    public void addRecord(){
        realm.beginTransaction();

        Number num = realm.where(Curso.class).max("ID");
        int nextId = (num==null) ? 1:num.intValue()+1;
            Curso curso = realm.createObject(Curso.class,nextId);



       curso.setDescricao(Descrição.getText().toString());
    curso.setNome(nomedocurso.getText().toString());

        realm.commitTransaction();
    }
    public void updateRecord(){
        RealmResults<Curso> results = realm.where(Curso.class).equalTo("ID", Integer.parseInt(iddocurso.getText().toString())).findAll();

        realm.beginTransaction();

        for(Curso curso : results) {
            curso.setDescricao(Descrição.getText().toString());
            curso.setNome(nomedocurso.getText().toString());
        }
        realm.commitTransaction();
    }
    public void deleteRecord(){
        RealmResults<Curso> results = realm.where(Curso.class).equalTo("ID", Integer.parseInt(iddocurso.getText().toString())).findAll();

        realm.beginTransaction();

        results.deleteAllFromRealm();

        realm.commitTransaction();
    }

    public  void viewRecord(){
        RealmResults<Curso> results2 = realm.where(Curso.class).findAll();
        text.setText("");
        for(Curso curso : results2){
            text.append(curso.getID()+" " +curso.getDescricao() + " "+"\n");
        }
        if(results2.isEmpty()){
            text.append("não adicionou");
        }

    }

}
